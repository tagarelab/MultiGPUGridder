432 void gpuGridder::CASImgsToImgs(cudaStream_t &stream, float *CASImgs, float *Imgs, int numImgs, cufftComplex *CASImgsComplex)
433 {
434  // Convert CAS images to images using an inverse FFT
435  // CASImgs, Imgs, and CASImgsComplex, are the device allocated arrays (e.g. d_CASImgs) at some offset from the beginning of the array
436 
437  if (this->verbose == true)
438  {
439  std::cout << "gpuGridder::CASImgsToImgs() on GPU " << this->GPU_Device << '\n';
440  PrintMemoryAvailable();
441  }
442 
443  int CASImgSize = this->d_CASImgs->GetSize(0);
444  int ImgSize = this->d_Imgs->GetSize(0);
445 
446  // Has the inverse FFT been planned? If not create one now
447  if (this->inverseFFTImagesFlag == false)
448  {
449  cufftPlan2d(&this->inverseFFTImages, CASImgSize, CASImgSize, CUFFT_C2C);
450 
451  this->inverseFFTImagesFlag = true;
452  }
453 
454  // Convert the CASImgs to complex cufft type
455  std::unique_ptr<CASToComplexFilter> CASFilter(new CASToComplexFilter());
456  CASFilter->SetCASVolume(CASImgs);
457  CASFilter->SetComplexOutput(CASImgsComplex);
458  CASFilter->SetVolumeSize(CASImgSize);
459  CASFilter->SetNumberOfSlices(numImgs);
460  CASFilter->Update(&stream);
461 
462  // Run a FFTShift on each 2D slice
463  std::unique_ptr<FFTShift2DFilter<cufftComplex>> FFTShiftFilter(new FFTShift2DFilter<cufftComplex>());
464  FFTShiftFilter->SetInput(CASImgsComplex);
465  FFTShiftFilter->SetImageSize(CASImgSize);
466  FFTShiftFilter->SetNumberOfSlices(numImgs);
467  FFTShiftFilter->Update(&stream);
468 
469  // Set the FFT plan to the current stream to process
470  cufftSetStream(this->inverseFFTImages, stream);
471 
472  // Execute the inverse FFT on each 2D array
473  // cufftPlanMany is not feasible since the number of images changes and
474  // cufftDestroy is blocks the CPU and causes memory leaks if not called
475  // FFT on each 2D slice has similar computation speed as cufftPlanMany
476  for (int i = 0; i < numImgs; i++)
477  {
478  cufftExecC2C(this->inverseFFTImages,
479  &CASImgsComplex[i * CASImgSize * CASImgSize],
480  &CASImgsComplex[i * CASImgSize * CASImgSize],
481  CUFFT_INVERSE);
482  }
483 
484  // Run a 2D FFTShift again
485  FFTShiftFilter->SetInput(CASImgsComplex);
486  FFTShiftFilter->SetImageSize(CASImgSize);
487  FFTShiftFilter->SetNumberOfSlices(numImgs);
488  FFTShiftFilter->Update(&stream);
489 
490  // Extract the real component of the complex images
491  std::unique_ptr<ComplexToRealFilter> ComplexToReal(new ComplexToRealFilter());
492  ComplexToReal->SetComplexInput(CASImgsComplex);
493  ComplexToReal->SetRealOutput(CASImgs);
494  ComplexToReal->SetVolumeSize(CASImgSize);
495  ComplexToReal->SetNumberOfSlices(numImgs);
496  ComplexToReal->Update(&stream);
497 
498  // Crop the images to remove the zero padding
499  std::unique_ptr<CropVolumeFilter> CropFilter(new CropVolumeFilter());
500  CropFilter->SetInput(CASImgs);
501  CropFilter->SetInputSize(CASImgSize);
502  CropFilter->SetNumberOfSlices(numImgs);
503  CropFilter->SetOutput(Imgs);
504  CropFilter->SetCropX((CASImgSize - ImgSize) / 2);
505  CropFilter->SetCropY((CASImgSize - ImgSize) / 2);
506  CropFilter->SetCropZ(0);
507  CropFilter->Update(&stream);
508 
509  // Normalize for the scaling introduced during the FFT
510  float normalizationFactor = ImgSize * ImgSize * interpFactor * interpFactor;
511 
512  std::unique_ptr<DivideScalarFilter> DivideScalar(new DivideScalarFilter());
513  DivideScalar->SetInput(Imgs);
514  DivideScalar->SetScalar(float(normalizationFactor));
515  DivideScalar->SetVolumeSize(ImgSize);
516  DivideScalar->SetNumberOfSlices(numImgs);
517  DivideScalar->Update(&stream);
518 }

562  std::unique_ptr<FFTShift2DFilter<cufftComplex>> FFTShiftFilter(new FFTShift2DFilter<cufftComplex>());
563  FFTShiftFilter->SetInput(CASImgsComplex);
564  FFTShiftFilter->SetImageSize(CASImgSize);
565  FFTShiftFilter->SetNumberOfSlices(numImgs);
566  FFTShiftFilter->Update(&stream);
567

569  cufftSetStream(this->forwardFFTImages, stream);
570 
571  // Execute the forward FFT on each 2D array
572  // cufftPlanMany is not feasible since the number of images changes and
573  // cufftDestroy blocks the CPU and causes memory leaks if not called
574  // FFT on each 2D slice has similar computation speed as cufftPlanMany
575  for (int i = 0; i < numImgs; i++)
576  {
577  cufftExecC2C(this->forwardFFTImages,
578  &CASImgsComplex[i * CASImgSize * CASImgSize],
579  &CASImgsComplex[i * CASImgSize * CASImgSize],
580  CUFFT_FORWARD);
581  }
582 
583  // Run the 2D FFTShift again
584  FFTShiftFilter->SetInput(CASImgsComplex);
585  FFTShiftFilter->SetImageSize(CASImgSize);
586  FFTShiftFilter->SetNumberOfSlices(numImgs);
587  FFTShiftFilter->Update(&stream);
588

590  std::unique_ptr<ComplexToCASFilter> ComplexToCAS(new ComplexToCASFilter());
591  ComplexToCAS->SetComplexInput(CASImgsComplex);
592  ComplexToCAS->SetCASVolumeOutput(CASImgs);
593  ComplexToCAS->SetVolumeSize(CASImgSize);
594  ComplexToCAS->SetNumberOfSlices(numImgs);
595  ComplexToCAS->Update(&stream);

598 gpuGridder::Offsets gpuGridder::PlanOffsetValues(int coordAxesOffset, int nAxes, int numStreams)
599 {
600  // Loop through all of the coordinate axes and calculate the corresponding pointer offset values
601  // which are needed for running the CUDA kernels
602 
603  if (this->verbose == true)
604  {
605  std::cout << "gpuGridder::PlanOffsetValues()" << '\n';
606  }
607 
608  // For compactness define the CASImgSize, CASVolSize, and ImgSize here
609  int CASImgSize = this->d_CASImgs->GetSize(0);
610  int CASVolSize = this->d_CASVolume->GetSize(0);
611  int ImgSize = this->d_Imgs->GetSize(0);
612 
613  // Create an instance of the Offsets struct
614  gpuGridder::Offsets Offsets_obj;
615  Offsets_obj.num_offsets = 0;
616 
617  // Cumulative number of axes which have already been assigned to a CUDA stream
618  int processed_nAxes = 0;
619 
620  // While we have coordinate axes to process, loop through the GPUs and the streams
621  int MaxBatches = 500; // Maximum iterations in case we get stuck in the while loop for some reason
622  int batch = 0;
623 
624  // Initialize a variable which will remember how many axes have been assigned to the GPU
625  // during the current batch. This is needed for calculating the offset for the stream when
626  // the number of streams is greater than the number of GPUs. This resets to zeros after each
627  // batch since the same GPU memory is used for the next batch.
628  int numAxesGPU_Batch = 0;
629 
630  // Estimate how many coordinate axes to assign to each stream
631  int EstimatedNumAxesPerStream;
632  if (nAxes <= this->MaxAxesToAllocate)
633  {
634  // The number of coordinate axes is less than or equal to the total number of axes to process
635  EstimatedNumAxesPerStream = ceil((double)nAxes / (double)numStreams);
636  }
637  else
638  {
639  // Several batches will be needed so evenly split the MaxAxesToAllocate by the number of streams
640  EstimatedNumAxesPerStream = ceil((double)this->MaxAxesToAllocate / (double)numStreams);
641  }
642 
643  if (this->verbose == true)
644  {
645  std::cout << "gpuGridder::PlanOffsetValues() "
646  << "estimated number of axes to process on each stream = " << EstimatedNumAxesPerStream << '\n';
647 
648  std::cout << "gpuGridder::PlanOffsetValues() "
649  << "number of axes to process on this GPU " << nAxes << '\n';
650 
651  std::cout << "gpuGridder::PlanOffsetValues() "
652  << "number of streams for this GPU " << numStreams << '\n';
653  }
654 
655  while (processed_nAxes < nAxes && batch < MaxBatches)
656  {
657  for (int i = 0; i < numStreams; i++) // Loop through the streams
658  {
659  // Make sure we have enough memory allocated to process this stream on the current batch
660  if (numAxesGPU_Batch + EstimatedNumAxesPerStream > this->MaxAxesToAllocate)
661  {
662  continue;
663  }
664 
665  std::cout << "processed_nAxes: " << processed_nAxes << " out of " << nAxes << '\n';
666 
667  // Have all the axes been processed?
668  if (processed_nAxes >= nAxes)
669  {
670  Offsets_obj.numAxesPerStream.push_back(0);
671 
672  std::cout << "Offsets_obj.numAxesPerStream.push_back(0)" << '\n';
673  continue;
674  }
675 
676  // Save the current batch number
677  Offsets_obj.currBatch.push_back(batch);
678 
679  if (batch > Offsets_obj.Batches)
680  {
681  Offsets_obj.Batches = batch;
682  }
683 
684  // If we're about to process more than the number of coordinate axes, process the remaining faction of numAxesPerStream
685  if (processed_nAxes + EstimatedNumAxesPerStream >= nAxes)
686  {
687  // Process the remaining fraction of EstimatedNumAxesPerStream
688  Offsets_obj.numAxesPerStream.push_back(std::min(EstimatedNumAxesPerStream, nAxes - processed_nAxes));
689  }
690  else
691  {
692  // Save the estimated number of axes to the numAxesPerStream
693  Offsets_obj.numAxesPerStream.push_back(EstimatedNumAxesPerStream);
694  }
695 
696  // Calculate the offsets (in bytes) to determine which part of the array to copy for this stream
697  // When using multiple GPUs coordAxesOffset will be the number already assigned to other GPUs
698  Offsets_obj.CoordAxes_CPU_Offset.push_back((processed_nAxes + coordAxesOffset) * 9); // Each axes has 9 elements (X, Y, Z)
699  Offsets_obj.coord_Axes_CPU_streamBytes.push_back(Offsets_obj.numAxesPerStream.back() * 9 * sizeof(float));
700 
701  // Use the number of axes already assigned to this GPU since starting the current batch to calculate the currect offset
702  Offsets_obj.gpuCASImgs_Offset.push_back(numAxesGPU_Batch * CASImgSize * CASImgSize);
703  Offsets_obj.gpuImgs_Offset.push_back(numAxesGPU_Batch * ImgSize * ImgSize);
704  Offsets_obj.gpuCoordAxes_Stream_Offset.push_back(numAxesGPU_Batch * 9);
705 
706  // Optionally: Copy the resulting CAS images back to the host pinned memory (CPU)
707  // Have to use unsigned long long since the array may be longer than the max value int32 can represent
708  // imgSize is the size of the zero padded projection images
709  unsigned long long *CASImgs_Offset = new unsigned long long[3];
710  CASImgs_Offset[0] = (unsigned long long)(CASImgSize);
711  CASImgs_Offset[1] = (unsigned long long)(CASImgSize);
712  CASImgs_Offset[2] = (unsigned long long)(processed_nAxes + coordAxesOffset);
713 
714  Offsets_obj.CASImgs_CPU_Offset.push_back(CASImgs_Offset[0] * CASImgs_Offset[1] * CASImgs_Offset[2]);
715 
716  // How many bytes are the output images?
717  Offsets_obj.gpuCASImgs_streamBytes.push_back(CASImgSize * CASImgSize * Offsets_obj.numAxesPerStream.back() * sizeof(float));
718 
719  // Have to use unsigned long long since the array may be longer than the max value int32 can represent
720  // imgSize is the size of the zero padded projection images
721  unsigned long long *Imgs_Offset = new unsigned long long[3];
722  Imgs_Offset[0] = (unsigned long long)(ImgSize);
723  Imgs_Offset[1] = (unsigned long long)(ImgSize);
724  Imgs_Offset[2] = (unsigned long long)(processed_nAxes + coordAxesOffset);
725 
726  Offsets_obj.Imgs_CPU_Offset.push_back(Imgs_Offset[0] * Imgs_Offset[1] * Imgs_Offset[2]);
727 
728  // How many bytes are the output images?
729  Offsets_obj.gpuImgs_streamBytes.push_back(ImgSize * ImgSize * Offsets_obj.numAxesPerStream.back() * sizeof(float));
730 
731  // Update the overall number of coordinate axes which have already been assigned to a CUDA stream
732  processed_nAxes = processed_nAxes + Offsets_obj.numAxesPerStream.back();
733 
734  // Update the number of axes which have been assigned to this GPU during the current batch
735  numAxesGPU_Batch = numAxesGPU_Batch + Offsets_obj.numAxesPerStream.back();
736 
737  // Add one to the number of offset values
738  Offsets_obj.num_offsets++;
739 
740  // Remember which stream this is
741  Offsets_obj.stream_ID.push_back(i);
742  }
743 
744  // Increment the batch number
745  batch++;
746 
747  // Reset the number of axes processed during the current batch variable
748  numAxesGPU_Batch = 0;
749  }
750 
751  return Offsets_obj;
752 }
